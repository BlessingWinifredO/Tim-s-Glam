// IMPROVED FIRESTORE RULES - Review and compare with your current rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper function to check if user is an admin
    // NOTE: Requires custom claims set via Admin SDK
    function isAdmin() {
      return request.auth.token.admin == true;
    }

    // Helper to check if user owns this document
    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Products collection - admins only for write
    match /products/{productId} {
      allow read: if true; // Public read
      allow create: if request.auth != null && isAdmin(); // Only admins
      allow update: if request.auth != null && isAdmin(); // Only admins
      allow delete: if request.auth != null && isAdmin(); // Only admins
    }

    // Orders collection - customers own their orders, admins can read all
    match /orders/{orderId} {
      allow read: if request.auth != null && (
        isOwner(resource.data.userId) || isAdmin()
      );
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId; // Customer creates own order
      allow update: if request.auth != null && isAdmin(); // Only admin can update order status
      allow delete: if request.auth != null && isAdmin(); // Only admin can delete
    }

    // App settings - admins only
    match /appSettings/{settingId} {
      allow read: if true; // Public read for settings
      allow write: if request.auth != null && isAdmin(); // Only admins
    }

    // Users collection - CRITICAL SECURITY
    match /users/{userId} {
      // Public read - anyone can view profiles
      allow read: if true;
      
      // User can create only their own profile
      allow create: if request.auth != null && isOwner(userId);
      
      // User can update ONLY their own profile (ADMIN CANNOT MODIFY CUSTOMER PROFILES)
      allow update: if request.auth != null && isOwner(userId);
      
      // Special case: Unauthenticated email verification (for verification flow)
      allow update: if request.auth == null
        && resource.data.emailVerified != true
        && request.resource.data.diff(resource.data).changedKeys().hasOnly(['emailVerified'])
        && request.resource.data.emailVerified == true;
      
      // User can delete only their own profile
      allow delete: if request.auth != null && isOwner(userId);
    }

    // Verification codes - authenticated users only
    match /verificationCodes/{email} {
      allow create: if request.auth != null; // Any authenticated user can create
      allow read, update: if request.auth != null; // Authenticated users can read/update
      allow delete: if request.auth != null; // Authenticated users can delete
    }

    // Password reset codes - open but could be restricted
    match /resetCodes/{email} {
      allow read, write: if true; // Currently open (consider restricting)
    }

    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
